package fr.univsavoie.java.swing.tortue;

import java.awt.BorderLayout;
import javax.swing.JFrame;
import javax.swing.JPanel;

/**
* Une JFrame qui crée une tortue et les éléments permettant de la manipuler.
*
* @author Stéphane Talbot
*/
public class TortueFrame extends JFrame {

    /**
     * Création d'une JFrame pour afficher et manipuler une tortue.
     *
     * @param t
     *            la tortue à afficher...
     *
     * @see Tortue
     * @see TortuePanel
     */
    public TortueFrame(Tortue t) {
        JPanel afficheur = new TortuePanel(t);
        this.getContentPane().setLayout(new BorderLayout(2, 2));
        this.getContentPane().add(afficheur, BorderLayout.CENTER);
        // on indique qu'on veut quitter l'application à la fermeture de la frame
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
    }

    /**
     * Constructeur sans paramètre pour la JFrame d'affichage de la tortue.
     * Il permet de construire une JFrame qui crée une tortue (coordonnées (50,50) et
     * orientation EST) ainsi que les éléments permettant de manipuler cette tortue.
     *
     * @see Tortue
     * @see TortuePanel
     *
     */
    public TortueFrame() {
        this(new Tortue(50, 50, Tortue.EST));
    }

    /**
     * La méthode main sert juste à lancer l'application. <br>
     * Création d'une tortue (coordonnées (100,50) et orientation SUD). <br>
     * Création d'une JFrame, instance de TortueFrame et affichage de celle-ci.
     *
     * @param args
     *            java.lang.String[]
     *
     * @see java.awt.Component#setVisible(boolean)
     */
    public static void main(String args[]) {
        Tortue caroline = new Tortue(100, 50, Tortue.SUD);
        JFrame f = new TortueFrame(caroline);
        f.setSize(300, 300);
        f.setVisible(true);
    }
}
