package fr.univsavoie.java.swing.tortue;

import java.awt.event.*;

/**
* Un KeyListener pour manipuler une tortue avec le clavier.
*
* La fleche vers le haut permet de faire avancer la tortue. La fleche vers la
* droite permet de faire tourner la tortue à droite. La fleche vers la gauche
* permet de faire tourner la tortue à gauche.
*
* @see Tortue
*
* @author Stéphane Talbot
*/
public class TortueKeyListener extends KeyAdapter {
    /**
     * La tortue que l'on veut manipuler
     */
    private Tortue tortue;

    /**
     * Construit un KeyListeneur pour manipuler une tortue à l'aide du clavier.
     *
     * @param t
     *            la tortue à manipuler...
     */
    public TortueKeyListener(Tortue t) {
        this.tortue = t;
    }

    /**
     * On a appuyé sur une touche du clavier.
     *
     * Si on a appuyé sur :
     * <ul>
     * <li>la fleche vers le haut, on fait avancer la tortue.</li>
     * <li>la fleche vers la droite, on fait tourner la tortue à droite.</li>
     * <li>la fleche vers la gauche, on fait faire tourner la tortue à gauche.</li>
     * </ul>
     *
     * @param e
     *            KeyEvent
     *
     * @see Tortue
     */
    public void keyPressed(KeyEvent e) {
        switch (e.getKeyCode()) {
        case KeyEvent.VK_UP:
            this.getTortue().avancer();
            break;
        case KeyEvent.VK_LEFT:
            this.getTortue().tournerGauche();
            break;
        case KeyEvent.VK_RIGHT:
            this.getTortue().tournerDroite();
            break;
        }
    }

    /**
     * Renvoie la tortue à manipuler.
     *
     * @return la tortue associée au listener.
     *
     * @see Tortue
     */
    private Tortue getTortue() {
        return this.tortue;
    }
}
