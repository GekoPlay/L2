package fr.univsavoie.java.swing.tortue;

import java.awt.event.MouseAdapter;
import java.awt.event.MouseListener;
import java.awt.event.MouseEvent;

/**
* Un MouseListener pour manipuler une tortue avec la souris.
*
* Un clic permet de déplacer la tortue en direction du pointeur de la souris.
*
* @see Tortue
*
* @author Stéphane Talbot
*/
public class TortueMouseListener extends MouseAdapter {
    /**
     * La tortue que l'on veut manipuler
     */
    private Tortue tortue;

    /**
     * Construit un MouseListeneur pour manipuler une tortue à l'aide de la
     * souris.
     *
     * @param t
     *            la tortue à manipuler...
     */
    public TortueMouseListener(Tortue t) {
        this.tortue = t;
    }

    /**
     * On a appuyé sur un bouton de la souris.
     *
     * Suivant les positions relatives de la tortue et du pointeur de la souris,
     * la tortue va avancer ou tourner en direction du pointeur de la souris.
     *
     * @param e
     *            MouseEvent
     *
     * @see Tortue
     */
    public void mousePressed(MouseEvent e) {
        // juste pour donner le focus au composant une fois qu'on a cliqué
        e.getComponent().requestFocus();
       
        int mouseX = e.getX();
        int mouseY = e.getY();
        int tX = this.tortue.getX();
        int tY = this.tortue.getY();
        int tDir = this.tortue.getDirection();
        int dx = mouseX - tX;
        int dy = mouseY - tY;

        switch (tDir) {
        case Tortue.SUD:
            if (dy > 0) {
                this.getTortue().avancer();
            } else if (dx > 0) {
                this.getTortue().tournerGauche();
            } else {
                this.getTortue().tournerDroite();
            }
            break;
        case Tortue.NORD:
            if (dy < 0) {
                this.getTortue().avancer();
            } else if (dx < 0) {
                this.getTortue().tournerGauche();
            } else {
                this.getTortue().tournerDroite();
            }
            break;
        case Tortue.EST:
            if (dx > 0) {
                this.getTortue().avancer();
            } else if (dy < 0) {
                this.getTortue().tournerGauche();
            } else {
                this.getTortue().tournerDroite();
            }
            break;
        case Tortue.OUEST:
            if (dx < 0) {
                this.getTortue().avancer();
            } else if (dy > 0) {
                this.getTortue().tournerGauche();
            } else {
                this.getTortue().tournerDroite();
            }
            break;
        }
    }

    /**
     * Renvoie la tortue à manipuler.
     *
     * @return la tortue associée au listener.
     *
     * @see Tortue
     */
    private Tortue getTortue() {
        return this.tortue;
    }
}
